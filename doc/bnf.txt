Program        := Statement*
Statement      := Expr;
Expr           := AssignmentExpr
AssignmentExpr := AddExpr
               := AddExpr = AssignmentExpr
AddExpr        := MulExpr
               := AddExpr + MulExpr
               := AddExpr - MulExpr
MulExpr        := PrimaryExpr
               := MulExpr * PrimaryExpr
               := MulExpr / PrimaryExpr
PrimaryExpr    := Num
               := (Expr)
               := Identifier
